@for $i from 0 through 12 {
	.f-#{$i} {
		flex: $i;
	}
}

// 定义字体(rpx)单位，10~100rpx  fs-10 ~ fs-100
@for $i from 10 through 100 {
	.fs-#{$i} {
		font-size: $i + rpx;
	}
}

// 定义内外边距，历遍1-100
@for $i from 0 through 100 {
	// 得出：m-all-30
	.m-all-#{$i} {
		margin: $i + rpx;
	}
	
	// 得出：p-all-30
	.p-all-#{$i} {
		padding: $i + rpx;
	}
	
	@each $short,$long in l left, t top, r right, b bottom {
		// 缩写版，结果如： ml-30
		// 定义外边距
		.m#{$short}-#{$i} {
			margin-#{$long}: $i + rpx;
		}
		
		// 定义内边距
		.p#{$short}-#{$i} {
			padding-#{$long}: $i + rpx;
		}
	}
}

//全局宽高
